# =============================================================================
# Development Usage Notes
# =============================================================================
# 
# Installation:
# pip install -r requirements-dev.txt
# 
# Code Quality Commands:
# black . --check              # Check code formatting
# black .                      # Format code
# isort . --check-only         # Check import sorting
# isort .                      # Sort imports
# flake8 .                     # Lint code
# mypy .                       # Type checking
# bandit -r .                  # Security scanning
# safety check                 # Dependency vulnerability check
# 
# Testing Commands:
# pytest                       # Run all tests
# pytest --cov=.              # Run tests with coverage
# pytest -v                   # Verbose test output
# pytest -x                   # Stop on first failure
# pytest tests/unit/          # Run unit tests only
# pytest tests/integration/   # Run integration tests only
# 
# Database Commands:
# pgcli postgresql://user:pass@localhost:5432/dbname  # Connect to PostgreSQL
# redis-cli -h localhost -p 6379                     # Connect to Redis
# 
# Pre-commit Setup:
# pre-commit install           # Install git hooks
# pre-commit run --all-files  # Run hooks on all files
# 
# Documentation:
# sphinx-build -b html docs docs/_build  # Generate HTML documentation
# 
# Profiling:
# python -m line_profiler script.py.lprof  # Line profiling
# python -m memory_profiler script.py      # Memory profiling
# py-spy record -o profile.svg -- python main.py  # Sampling profiler
# =============================================================================
